1. create `src/.env`
2. `yarn add dotenv`
3. edit `index.ts`
```typescript
//...
dotenv.config();
//console.log(process.env)
// you can see logs when executing `yarn dev`
//...
```
4. edit `ormConfig.ts`

```typescript
const connectionOptions: ConnectionOptions = {
	//...
	host: process.env.DB_ENDPOINT,
	username: process.env.DB_USERNAME,
	password: process.env.DB_PASSWORD
}
```
